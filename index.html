// Core formula — determines system stability
const rho = lambda / (k * mu);
// ρ = λ / (k · μ)

λ = arrival rate (req/s)
μ = service rate per server
k = number of active servers
02 CPU EWMA Model Exponential Smoothing
// Smooths CPU spikes — prevents noisy auto-scaling decisions
const ALPHA = 0.3; // smoothing factor
const noise = gaussianNoise(0, 3.5);

cpu[t] = ALPHA * rho[t] * 100
+ (1 - ALPHA) * cpu[t-1]
+ noise;
α=0.3 weights recent ρ heavily while damping historical variance
